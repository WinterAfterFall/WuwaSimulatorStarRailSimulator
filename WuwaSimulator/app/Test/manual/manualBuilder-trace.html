app/manualBuilder.ts — mergeQueues(queue1, queue2) 
function mergeQueues(...queues): Queue<RotationAction> {
    const merged = new Queue<RotationAction>();
    for (const q of queues) {
        while (!q.isEmpty()) merged.enqueue(q.dequeue()!);
    }
    return merged;
}

app/Simulator/RotationDirector.ts — step() — วนทุก action
แต่ละ action: check lock → nextAction() → execute() → รัน tick() จนปลด lock
private step(): boolean {
    // ① ถ้า GlobalLock ค้างอยู่ (จาก action ก่อน) → หยุด
    if (this.timeline.isGlobalLocked) return false;

    // ② ดึง action ถัดไป
    const action = this.nextAction();
    if (!action) return false; // หมด action → จบ

    // ③ execute fn ใน RotationAction → schedule event ลง timeline
    action.execute();

    // ④ รัน tick จนปลด lock (หรือ timeline ว่าง)
    do {
        this.timeline.tick();
    } while (this.timeline.isGlobalLocked && !this.timeline.isEmpty);

    return true;
}

app/Simulator/RotationDirector.ts — nextAction()
setupQueue ก่อน (dequeue) → หมดแล้วใช้ loopQueue (rotate วน) จนครบ maxLoops
private nextAction(): RotationAction | undefined {
    if (!this.setupQueue.isEmpty()) {
        return this.setupQueue.dequeue(); // FIFO ก่อน
    }
    if (this.loopQueue.isEmpty()) return undefined;
    if (this.currentLoopCount >= this.maxLoops) return undefined; // ครบแล้ว

    const action = this.loopQueue.rotate(); // dequeue + enqueue ท้าย
    this.loopStepCount++;
    if (this.loopStepCount === this.loopQueue.length) { // ครบ 1 รอบ (6 actions)
        this.loopStepCount = 0;
        this.currentLoopCount++;
    }
    return action;
}

export class AttackActionEvent extends ActionEvent {
    private readonly onExecute?: () => void; // ← ใหม่

    constructor(
        name, time, duration, unit, actionType, isManual,
        priority = 0,
        onExecute?: () => void  // ← param ใหม่
    ) {
        super(name, time, duration, unit, actionType, isManual, priority);
        this.onExecute = onExecute;
    }

    public execute(): void {
        this.unit.setBusy();    // unit.actionState = Busy
        this.onExecute?.();    // → console.log("[f0] Test1-Ult")
    }
}

pop event → ขยับ currentFrame → execute() → จัดการ GlobalLock จาก ActionEvent / NotificationEvent
public tick(): CombatEvent | undefined {
    const event = this.ipq.pop(); // ดึง event frame น้อยสุด
    if (!event) return undefined;

    this.currentFrame = event.time; // ← ขยับ frame!
    event.execute();               // ← รัน event

    // ── Lock management ──
    if (event instanceof ActionEvent && event.isManual) {
        this.isGlobalLocked = true;  // AttackActionEvent isManual=true → lock
    }
    if (event instanceof NotificationEvent) {
        if (event.notifyType === NotificationType.ChangeToAuto) {
            this.isGlobalLocked = false; // ปลด lock แต่ unit ยัง Busy
        }
        if (event.notifyType === NotificationType.EndAction) {
            this.isGlobalLocked = false; // ปลด lock
            event.unit?.setFree();        // unit.actionState = Free
        }
    }
    return event;
}
