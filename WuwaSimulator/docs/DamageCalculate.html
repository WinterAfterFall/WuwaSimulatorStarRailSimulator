DamageCalculate.ts

ระบบคำนวณ damage สำหรับ Wuthering Waves Simulator — อธิบายทุก function ทุก formula

ภาพรวมระบบ

ไฟล์นี้เป็น service หลักสำหรับคำนวณ damage ในการ simulate การต่อสู้ รับ Damage object เป็น input ซึ่งประกอบด้วย attacker, target list, element, skill multipliers แล้วคำนวณ damage ออกมาต่อ target แต่ละตัว

หลักการทำงานหลักแบ่งเป็น 3 ขั้น:

  1. Aggregate stats — รวม stat จากทุก dimension (base, per-action, per-element)
  2. Compute sides — คำนวณค่า derived stats ของ attacker และ target แยกกัน
  3. Apply formula — คูณ multiplier ทุกตัวเข้าด้วยกัน แล้ว print ผลลัพธ์

function sumStat() helper

เป็นหัวใจหลักของระบบ — รวมค่า stat จาก ทุก dimension ที่เกี่ยวข้องกับ skill นั้น ก่อนหน้านี้ code เดิมอ่าน stat แบบ hardcode ทำให้พลาด buff ที่ผูกกับ action type หรือ element function นี้แก้ปัญหานั้นโดยการวน loop อัตโนมัติ 

function sumStat(unit: Unit, st: StatsType, damage: Damage): number {
    const { attackTypeList, element } = damage;

    // Phase 1 — base + per-action dimension
    let total = unit.getStats(st);               // key: "Dmg Bonus"
    for (const at of attackTypeList) {
        total += unit.getStats(st, at);           // key: "Dmg Bonus-Skill"
    }

    // Phase 2 — element dimension (เฉพาะ Dmg, Amp, Respen, ElemRed)
    if (ELEMENT_DIMENSION_STATS.has(st) && element !== ElementType.None) {
        total += unit.getStats(st, element, ActionType.None); // "Dmg Bonus-Spectro-None"
        for (const at of attackTypeList) {
            total += unit.getStats(st, element, at);           // "Dmg Bonus-Spectro-Skill"
        }
    }

    return total;
}

Phase 1 — Base + Per-action dimension

ทุก stat ผ่าน phase นี้เสมอ สูตรคือ:

total = get(stat) ← global buff (ไม่ขึ้นกับ action)
+ get(stat, BA) ← buff เฉพาะตอนใช้ Basic Attack
+ get(stat, Skill) ← buff เฉพาะตอนใช้ Resonance Skill
+ get(stat, ...) ← วน loop ทุก actionType ใน attackTypeList

Phase 2 — Element dimension (เฉพาะบาง stat)

stats ที่อยู่ใน ELEMENT_DIMENSION_STATS จะได้รับการวน dimension element เพิ่มเติม เหตุผลคือ buff เหล่านี้ใน WuWa สามารถผูกกับ element ได้ เช่น Spectro DMG Bonus +15%, Spectro Skill DMG +10%

total += get(stat, Spectro, None) ← "Dmg Bonus-Spectro-None" (element-only buff)
+ get(stat, Spectro, BA) ← "Dmg Bonus-Spectro-BA"
+ get(stat, Spectro, Skill) ← "Dmg Bonus-Spectro-Skill"

✅ stats ที่ใช้ element dimension

  • StatsType.Dmg — DMG Bonus
  • StatsType.Amp — Amplify
  • StatsType.Respen — RES Penetration
  • StatsType.ElemRed — Elemental Reduction

❌ stats ที่ไม่ใช้ element dimension

  • StatsType.AtkP / FlatAtk
  • StatsType.CR / CD
  • StatsType.DefRed
  • StatsType.DmgRed
ทำไม ElemRed ต้องอยู่ใน ELEMENT_DIMENSION_STATS? ใน key system ของโปรเจกต์นี้ enemy resistance เก็บเป็น key "Elemental Reduction-Glacio-None" ซึ่งเป็น 3-argument form — ถ้าไม่รวม element dimension ใน sumStat จะอ่านค่าไม่เจอ เพราะ getStats(ElemRed) (1-arg) จะได้ 0 เสมอ

function computeAttackerStats()

รับ AllyUnit และ Damage แล้วคำนวณ stat ที่จำเป็นทั้งหมดของฝั่ง attacker เรียก sumStat() สำหรับแต่ละ stat type เพื่อให้ได้ค่าที่รวม buff ทุก dimension แล้ว 

function computeAttackerStats(attacker: AllyUnit, damage: Damage): AttackerStats {
    const atkP    = sumStat(attacker, StatsType.AtkP,    damage);
    const flatAtk = sumStat(attacker, StatsType.FlatAtk, damage);
    // ... (hp, def เหมือนกัน)

    return {
        level   : attacker.level,
        atk     : attacker.baseAtk * (1 + atkP) + flatAtk,
        hp      : attacker.baseHp  * (1 + hpP)  + flatHp,
        def     : attacker.baseDef * (1 + defP) + flatDef,
        cr      : Math.min(sumStat(attacker, StatsType.CR,       damage), 1),
        cd      : sumStat(attacker, StatsType.CD,       damage),
        dmgBonus: sumStat(attacker, StatsType.Dmg,      damage),
        amp     : sumStat(attacker, StatsType.Amp,      damage),
        defShred: sumStat(attacker, StatsType.DefShred, damage),
        respen  : sumStat(attacker, StatsType.Respen,   damage),
    };
}

function computeTargetStats()

รับ EnemyUnit และ Damage แล้วคำนวณ stat ฝั่ง enemy ที่กระทบต่อ damage ที่รับ เรียก sumStat() เหมือนกัน ทำให้รองรับกรณีที่ enemy มี buff หรือ debuff ที่ผูกกับ action type ได้ 

function computeTargetStats(target: EnemyUnit, damage: Damage): TargetStats {
    const defRed  = sumStat(target, StatsType.DefRed,  damage);
    const flatDef = sumStat(target, StatsType.FlatDef, damage);

    // baseDef จาก level, DefRed เป็น debuff บน enemy — ลด DEF ก่อนเข้าสูตร %DEF
    const baseDef = 8 * target.level + 792;
    const def     = baseDef * Math.max(0, 1 - defRed) + flatDef;

    return {
        def,
        dmgBonus: sumStat(target, StatsType.Dmg,    damage),
        dmgRed  : sumStat(target, StatsType.DmgRed, damage),
        elemRed : target.baseElemRed + sumStat(target, StatsType.ElemRed, damage),
    };
}

dmgBonussumStat(Dmg)สำหรับ enemy ที่มี self-buff (มักเป็น 0)
dmgRedsumStat(DmgRed)ลด damage ขาเข้าเป็น %

function applyDamageFormula()

รับ computed stats ของทั้ง 2 ฝั่ง แล้วคำนวณ damage ตาม WuWa formula ผลลัพธ์เป็น expected damage (crit ใช้ค่าเฉลี่ย CR × CD ไม่ใช่ RNG) 

const base = attackerStats.atk * m[MultiplierType.Atk]
           + attackerStats.hp  * m[MultiplierType.Hp]
           + attackerStats.def * m[MultiplierType.Def]
           + m[MultiplierType.Const];

const dmgBonusMulti  = 1 + attackerStats.dmgBonus;
const critMulti      = damage.isCritable ? 1 + attackerStats.cr * attackerStats.cd : 1;
const ampMulti       = 1 + attackerStats.amp;
const lvlFactor  = 800 + 8 * attackerStats.level;
const defMulti   = lvlFactor / (lvlFactor + targetStats.def * Math.max(0, 1 - attackerStats.defShred));
const resTotal   = targetStats.elemRed - attackerStats.respen;
const resMulti   = resTotal < 0   ? 1 - resTotal / 2
                 : resTotal < 0.8 ? 1 - resTotal
                 :                  1 / (1 + 5 * resTotal);
const reductionMulti = Math.max(0, 1 - targetStats.dmgRed);

return base * dmgBonusMulti * critMulti * ampMulti * defMulti * resMulti * reductionMulti;

อธิบายแต่ละ multiplier

⚔️

Base DMG

คำนวณจาก Damage.multipliers (SkillScaling) × stat แต่ละตัว สูตร: ATK×m.atk + HP×m.hp + DEF×m.def + m.const skill ที่ scale จาก ATK 100% จะมี m.atk = 1.0, ส่วน m.hp / m.def / m.const = 0

📈

DMG Bonus Multiplier = 1 + dmgBonus

รวม DMG Bonus ทุก dimension — general + element-specific + action-specific + element+action ตัวอย่าง: Spectro DMG +15% จะถูกรวมเข้ามา ถ้า damage.element = Spectro

🎯

Crit Multiplier = 1 + CR × CD

ใช้ค่าเฉลี่ย (expected value) ไม่ใช่ RNG CR ถูก cap ที่ 100% ใน computeAttackerStats() ถ้า damage.isCritable = false → multiplier = 1 เสมอ

⚡

Amplify Multiplier = 1 + amp

Amplify (Attribute Amplification ในเกม) รวม element+action dimension ตัวอย่างใช้งาน: Resonance Skill ที่มี Amplify buff จะเพิ่ม damage อีกชั้น

🛡️

DEF Multiplier = (800+8×LVL) / (800+8×LVL + DEF × (1−defShred))

WuWa DEF formula ตาม level จริงของ attacker — lvlFactor = 800 + 8 × attacker.level enemy DEF มาจาก 8×enemy.level+792 หลังจากลด DefRed (debuff บน enemy) แล้ว defShred คือ DEF Ignore ฝั่ง attacker ที่ใช้ใน formula (ไม่ใช่ debuff)

🔮

RES Multiplier — piecewise ตาม RES_total = elemRed − respen

elemRed = enemy.baseElemRed + sumStat(ElemRed) / respen = RES Pen จาก attacker RES_total ติดลบได้ (เมื่อ respen > elemRed) ทำให้ damage เพิ่มขึ้นได้:

  • RES < 0 → 1 − RES/2  (damage bonus แต่ diminishing)
  • 0 ≤ RES < 0.8 → 1 − RES  (linear reduction)
  • RES ≥ 0.8 → 1/(1+5×RES)  (hard cap zone)
🪨

Damage Reduction = max(0, 1 − dmgRed)

ลด damage ขาเข้าของ enemy เป็น % สุดท้าย max(0, ...) กัน dmgRed > 100% ทำให้ damage ติดลบ

export calculateDamage() public API

เป็น function เดียวที่ export ออกไป — entry point ของทั้งระบบ รับแค่ Damage ตัวเดียว แล้วจัดการทุกอย่างภายใน 

export function calculateDamage(damage: Damage): void {
    const attacker      = damage.attacker;
    const attackerStats = computeAttackerStats(attacker, damage); // คำนวณครั้งเดียว

    for (const target of damage.targets) {               // วนทุก target
        const targetStats = computeTargetStats(target, damage);
        const result      = applyDamageFormula(damage, attackerStats, targetStats);

        console.log(`[Damage] ${attacker.name} ตี ${target.name}: ${result.toFixed(2)}`);
    }
}

ตัวอย่าง output:

[Damage] Rover ตี Nightmare Dreamless: 12483.50
[Damage] Rover ตี Impermanence Heron: 8921.33
⚠️ การใช้งาน ต้อง set damage.attacker, damage.targets[], damage.element, และ damage.multipliers ให้ครบก่อนเรียก ถ้า damage.targets ว่างเปล่า function จะไม่ print อะไรเลย (loop ไม่ทำงาน)

สูตรรวม (Wuthering Waves Damage Formula)

Damage = Base × DMGBonus × Crit × Amp × DEF × RES × Reduction
Base = ATK × m.atk + HP × m.hp + DEF × m.def + m.const
ATK = baseAtk × (1 + AtkP_sum) + FlatAtk_sum
DMGBonus = 1 + Σ(Dmg bonus ทุก dimension)
Crit = 1 + min(CR, 1) × CD ← expected value
Amp = 1 + Σ(Amplify ทุก dimension)
Enemy DEF = (8×LVL_enemy + 792) × max(0, 1 − DefRed) ← DefRed = debuff บน enemy
lvlFactor = 800 + 8 × LVL_attacker
DEF = lvlFactor / (lvlFactor + Enemy DEF × max(0, 1 − defShred)) ← defShred = DEF Ignore ฝั่ง attacker
RES_total = elemRed − respen ← ติดลบได้
RES = 1 − RES/2   (RES < 0)
= 1 − RES     (0 ≤ RES < 0.8)
= 1/(1+5×RES)  (RES ≥ 0.8)
Reduction = max(0, 1 − dmgRed)
